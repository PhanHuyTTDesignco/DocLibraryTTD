Class mà bạn muốn lưu
Examples
public class ModelSaveTest
{    
  [DataBaseUtilsPropertyNotUsed]
  public string? Prop1 { get ; set ; }// khi gắn DataBaseUtilsPropertyNotUsedAttribute thì propery này sẽ không được save
  public string? Prop2 { get ; set ; }
  public int Prop3 { get ; set ; }
  public double Prop4 { get ; set ; }
}


var datasave = new List<ModelSaveTest>()
{
  new ModelSaveTest() { Prop2 = "E", Prop3 = 1, Prop4 = 0.2 },
  new ModelSaveTest() { Prop2 = "E" },
  new ModelSaveTest() 
} ;
var save = DataBaseUtils.Save(datasave);
